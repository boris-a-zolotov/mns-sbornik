\itB Из второй фигуры можно собрать горизонтальную полоску ширины 3, которой очевидно можно замостить плоскость. Из первой же фигуры соберём «лесенку» (смотреть рисунок ниже): так как и верхний, и нижний её край имеет вид «на три клетки вправо–на клетку вверх», этой лесенкой можно замостить плоскость, прикладывая её к себе.

\begin{center}
	\tikz{ \foreach \x in {-2,0,2} {\begin{scope}[xshift = \x cm]
		\draw[thick] (0,0) -- ++(0.5,0) -- ++(0,1) -- ++(-2,0) -- ++(0,-1)
			-- ++(0.5,0) -- ++(0,0.5) -- ++(1,0) -- ++(0,-0.5);
		\draw[thick,rotate=180,yshift=-0.5 cm,xshift=2 cm]
			(0,0) -- ++(0.5,0) -- ++(0,1) -- ++(-2,0) -- ++(0,-1)
			-- ++(0.5,0) -- ++(0,0.5) -- ++(1,0) -- ++(0,-0.5);
	\end{scope}}
			\draw (-5,0.25) node{$\ldots$};
			\draw (3,0.25) node{$\ldots$};
	}
\end{center}
